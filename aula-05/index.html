        import pandas as pd

# Exemplo de dados
dados = {'valor': [10, 12, 14, 13, 15, 100, 20, 19, 25, 22, 30, 35, 200]}
df = pd.DataFrame(dados)

# Função para remover outliers com base no IQR
def remover_outliers(df, coluna):
    Q1 = df[coluna].quantile(0.25)
    Q3 = df[coluna].quantile(0.75)
    IQR = Q3 - Q1
    
    # Definir limites inferior e superior para os outliers
    limite_inferior = Q1 - 1.5 * IQR
    limite_superior = Q3 + 1.5 * IQR
    
    # Filtrando os dados para remover os outliers
    df_sem_outliers = df[(df[coluna] >= limite_inferior) & (df[coluna] <= limite_superior)]
    
    return df_sem_outliers

# Remover outliers
df_sem_outliers = remover_outliers(df, 'valor')

print("Dados originais:")
print(df)
print("\nDados após remoção de outliers:")
print(df_sem_outliers)
